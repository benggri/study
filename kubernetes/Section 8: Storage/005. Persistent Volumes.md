```
 ---------------------------------------------------------------------

 ---------------------------------------------------------------------
```

```yml
apiVersion: v1
kind: PersistentVolume
metadata: pv-vol
spec:
  accessMode:
    - ReadWriteOnce
  capacity:
    storage: 1Gi
  hostPath: # Local Directory
    path: /tmp/data
  awsElasticBlockStore: # 외부 저장소
    volumeID: <volume-id> 
    fsType: ext4
```
